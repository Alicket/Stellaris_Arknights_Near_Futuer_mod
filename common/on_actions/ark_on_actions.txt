# Triggers when the game starts
on_game_start = {
	events = {
		ark_civic.20	# 创建国家用于存放领袖
	}
}

on_game_start_country = {
	events = {
		ark_set.1			# 强制修改非明日方舟国家选择的mod内容
		ark_game_start.3		# 询问是否扮演明日方舟国家
		ark_origins.0		# 起源效果（明日方舟）
		ark_civic.1		# 商业联合（第一个国民理念）
		ark_civic.4		# 骑士精神
	}
}

empire_init_capital_planet = {
	events = {
		ark_game_start.1
	}
}

# Triggers when trying to open the diplomacy view for countries with custom diplomacy.
# Country scope
# This = target country (player)
# From = source country
on_custom_diplomacy = {
	events = {
		ark_rhodes_island.2
	}
}

# Planet has taken damage from orbital bombardment
# This = Planet
# From = Bombarder
on_planet_bombarded = {
	events = {
		ark_orbital_bombardment.0	# 源石轨道轰炸消耗源石
		ark_orbital_bombardment.1		# 源石轨道轰炸污染人口：15破坏度
	}
}

on_monthly_laser_country = {
	events = {
		ark_set.3		# 检查并重置领袖肖像
		ark_set.4
	}
}
