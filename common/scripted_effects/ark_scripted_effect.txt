#输入	解释				默认值
#NUM	循环次数			 20
#MAX	初始感染者最大人数	  10	
#理论感染人数：NUM/2
initialize_ark_infected_population = {
	create_species = {
		class = Arknight
		portrait="arknights"
		name="Terraman"
		plural="Terramen"
		adjective="Terraman"
		name_list="ark_01"
		traits = {
			trait = trait_latent_oripathy
			trait = trait_natural_engineers
			trait = trait_nomadic
			trait = trait_quarrelsome
			trait = trait_deviants
			trait = trait_infected_persons
		}
		
	}
	while = {
		count = $NUM|20$
		limit = {
			check_variable = {
				which = start_infected_population
				value <= $MAX_POP|10$
			}
		}
		random_list = {
			10 = {
				random_owned_pop = {
					limit = {
						is_same_species = owner.species
					}
					kill_pop = yes
				}
				change_variable = {
					which = start_infected_population
					value = 1
				}
			}
			10 = {

			}
		}
	}
	while = {
		count = start_infected_population
		capital_scope = {
			create_pop = {
				species = last_created_species
				ethos = {
					ethic = ethic_materialist
				}
			}
			last_created_pop = {
				set_citizenship_type = {
					type = citizenship_slavery
					country = root
				}
			}
		}
	}
}
#输入			解释			备注
#COUNTRY_NAME 	国家本地化key	<string>必要
#ICON_FILE		国家图标		<string>必要，默认为hypergryph文件夹，无需添加.dds后缀。
#FLEET_NAME		舰队本地化key	<string>必要
#NUM_SHIPS		舰船数量		<int>默认为3
#COUNTRY_FLAG	添加国家flag	<string>非必要，不输入则不执行
#FLEET_FLAG		添加舰队flag	<string>非必要，不输入则不执行
#DISTANCE		选定舰队位置	<int>非必要，不输入则不执行。锚点为首都星系主恒星，极坐标定位中的距离
#ANGLE			选定舰队位置	<int>非必要，不输入则不执行。锚点为首都星系主恒星，极坐标定位中的角度
#输出
#母星系敌对舰队
initialize_ark_tragedy_birth_fleet = {
	owner_species = {
		save_event_target_as = ark_owner_species
	}
	create_country = {
		type = faction
		name = $COUNTRY_NAME$
		species = event_target:ark_owner_species
		name_list = root
		auto_delete = no
		flag = {
			icon = {
				category = hypergryph
				file = $ICON_FILE$.dds
			}
			background={
				category = backgrounds
				file = rounded_middle.dds
			}
			colors = {
				"black"
				"black"
				"null"
				"null"
			}
		}
		effect = {
			set_graphical_culture = root
			[[COUNTRY_FLAG] set_country_flag = $COUNTRY_FLAG$ ]
			establish_communications_no_message = root
			create_ship_design = {
				design = "NAME_Dagger"
			}
			create_fleet = {
				name = $FLEET_NAME$
				effect = {
					set_owner = prev
					[[FLEET_FLAG] set_fleet_flag = $FLEET_FLAG$ ]
					while = {
						count = $NUM_SHIPS|3$
						create_ship = {
							name = random
							design = last_created_design
						}
					}
					set_location = {
						target = root.capital_scope.solar_system.star
						[[DISTANCE] distance = $DISTANCE$ ]
						[[ANGLE] angle = $ANGLE$ ]
					}
				}
			}
		}
	}
}

#KB：精英领袖的领袖生成模板
ARK_K_APEX_born = {  
  from = {
	if = {
		limit = {
			owner = {
			    #days_passed > 30
			    check_variable = {
				  which = is_apex_$check_name$_spawn
				  value = 0
			    }					  
			}
		}
		set_name = $name$    #领袖姓名
		set_age = $age$      #领袖年龄
		set_is_female = $is_female|yes$     #领袖性别，不写就是女性
		set_skill = $skill_level$       #领袖技能等级
		change_leader_portrait = ARK$port$     #领袖肖像，仅需输入ARK后数字序号
		owner = {  #给这个领袖所属的国家打上变量标记领袖的刷新与否和存活与否
		   set_variable = { #生成标记
			which = is_apex_$check_name$_spawn
			value = 1
		   }
		   set_variable = { #存活标记
			which = is_apex_$check_name$_alive
			value = 1
		   }
		}	
			
	}

  }
	
}

		species = $species$
		name = $name$
		gender = $gender|female$  #不写就是女性

			[[trait4] trait=$trait4$]

			   set_variable = { #生成标记

				value = 1
			   }

				which = is_apex_$check_name$_alive

			   }
			   
			}	
		}
	}
