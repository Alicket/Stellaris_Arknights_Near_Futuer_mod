originiums_fire = {
	length = @EdictDuration
	icon = "GFX_edict_type_policy"

		upkeep = {

			multiplier = value:edict_size_effect
		}
	}
	
	potential = {
		OR = {

	}
	
	ai_weight = {

			is_at_war = yes
		}
	}
}
