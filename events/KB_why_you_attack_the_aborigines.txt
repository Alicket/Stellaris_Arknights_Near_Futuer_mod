country_event = {    #一个不显示的事件，根据玩家所处阶段对陆军登陆进行判决
	id = arknights_KWYATA.1
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		has_country_flag = the_aborigines_are_still_there #事件土著还在，而且进攻方是泰拉，这是最关键的
		fromfrom = {
			is_same_value = event_target:bombstar  #被登陆成功的是非塞特
		}
	}
	immediate = {
		remove_country_flag = the_aborigines_are_still_there #中断事件
		remove_country_flag = havent_found_aborigines_yet
		switch = {
			trigger = has_country_flag
			between_one_and_two = {
				country_event = {
					id = arknights_KWYATA.12
				}
			}
			between_two_and_three = {
				country_event = {
					id = arknights_KWYATA.23
				}
			}
			between_three_and_time = {
				country_event = {
					id = arknights_KWYATA.310
				}
			}
			between_two_and_four = {
				country_event = {
					id = arknights_KWYATA.24
				}
			}
			between_three_and_four = {
				country_event = {
					id = arknights_KWYATA.34
				}
			}
			between_four_and_five = {
				country_event = {
					id = arknights_KWYATA.45
				}
			}
			five_is_over = {
				country_event = {
					id = arknights_KWYATA.50
				}
			}
		}
	}

}
